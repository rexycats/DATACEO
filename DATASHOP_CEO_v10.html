📖 SQL Syntaxen

Overzicht van alle SQL-commando's die je nodig hebt als DataShop CEO.

SELECT Gegevens opvragen
SELECT kolom1, kolom2 FROM tabel WHERE conditie ORDER BY kolom ASC|DESC LIMIT n;
QueryResultaat
SELECT * FROM klantAlle klanten, alle kolommen
SELECT naam, stad FROM klantAlleen naam + stad kolommen
SELECT naam FROM klant WHERE actief = 1Namen van actieve klanten
SELECT naam FROM klant ORDER BY naam ASCAlfabetisch gesorteerd
SELECT * FROM product ORDER BY prijs DESC LIMIT 3Top 3 duurste producten
SELECT naam, email FROM klant WHERE stad = 'Gent'Klanten uit Gent
💡 * = alle kolommen. Beter specifieke kolommen noemen voor leesbaarheid. 💡 ASC = laag→hoog (standaard) · DESC = hoog→laag
WHERE Filteroperatoren
WHERE kolom = waarde -- exact gelijk WHERE kolom != waarde -- niet gelijk WHERE kolom > waarde -- groter dan WHERE kolom LIKE '%patroon%'-- bevat tekst WHERE kolom IN ('a','b','c') -- in lijst WHERE kolom BETWEEN x AND y -- bereik WHERE kolom IS NULL -- leeg veld WHERE c1 AND c2 -- beide voorwaarden WHERE c1 OR c2 -- één van beide
VoorbeeldBetekenis
WHERE stad = 'Gent'Exact Gent
WHERE prijs > 50Duurder dan €50
WHERE naam LIKE '%Jan%'Naam bevat "Jan"
WHERE stad IN ('Gent','Brussel')Gent of Brussel
WHERE prijs BETWEEN 10 AND 50€10 t.e.m. €50
WHERE actief = 1 AND stad = 'Gent'Actief én uit Gent
INSERT Nieuwe rij toevoegen
INSERT INTO tabel (kolom1, kolom2) VALUES (waarde1, waarde2);
QueryResultaat
INSERT INTO klant (naam, email, stad, actief)
VALUES ('Jana', '[email protected]', 'Gent', 1)		Voegt Jana toe als actieve klant in Gent
INSERT INTO product (naam, prijs, stock, categorie)
VALUES ('Lamp', 29.99, 10, 'Wonen')		Nieuw product "Lamp" voor €29.99 met 10 stuks
INSERT INTO review (klant_id, product_id, score, commentaar)
VALUES (1, 2, 5, 'Top product!')		Klant 1 geeft score 5 aan product 2
💡 Volgorde kolommen = volgorde waarden. Tekst staat tussen 'aanhalingstekens'. 💡 Geen klant_id nodig — AUTO_INCREMENT vult dit automatisch in.
UPDATE Bestaande rij(en) wijzigen
UPDATE tabel SET kolom = nieuwewaarde WHERE conditie;
QueryResultaat
UPDATE product SET prijs = 44.99 WHERE product_id = 2Wijzigt ALLEEN product 2 → nieuwe prijs €44.99
UPDATE klant SET actief = 0 WHERE klant_id = 4Deactiveert ALLEEN klant 4
UPDATE bestelling SET status = 'geleverd' WHERE bestelling_id = 1Markeert bestelling 1 als geleverd
UPDATE product SET stock = 20, prijs = 79.99 WHERE product_id = 5Meerdere kolommen tegelijk aanpassen
⚠️ ALTIJD WHERE! UPDATE klant SET actief = 0 zonder WHERE deactiveert ALLE klanten!
DELETE Rij(en) verwijderen
DELETE FROM tabel WHERE conditie;
QueryResultaat
DELETE FROM review WHERE score = 1Verwijdert alle 1-ster reviews
DELETE FROM bestelling WHERE datum < '2024-01-01'Verwijdert bestellingen van vóór 2024
DELETE FROM kortingscode WHERE actief = 0Ruimt inactieve kortingscodes op
⚠️ ALTIJD WHERE! DELETE FROM klant verwijdert ALLE klanten — onomkeerbaar! 💡 GDPR tip: gebruik liever UPDATE klant SET actief = 0 WHERE klant_id = x i.p.v. DELETE voor klantdata.
Aggregaten COUNT · SUM · AVG · MAX · MIN
COUNT(*) -- aantal rijen SUM(kolom) -- optelling AVG(kolom) -- gemiddelde MAX(kolom) -- grootste waarde MIN(kolom) -- kleinste waarde
QueryResultaat
SELECT COUNT(*) FROM klantHoeveel klanten zijn er totaal?
SELECT COUNT(*) FROM klant WHERE actief = 1Hoeveel actieve klanten?
SELECT AVG(prijs) FROM productGemiddelde productprijs
SELECT MAX(prijs) FROM productDuurste product
SELECT MIN(prijs), MAX(prijs) FROM productGoedkoopste én duurste
SELECT SUM(prijs * stock) FROM productTotale voorraadwaarde
GROUP BY Groeperen per categorie
SELECT kolom, COUNT(*) FROM tabel GROUP BY kolom HAVING COUNT(*) > n; -- filter op groep
QueryResultaat
SELECT klant_id, COUNT(*) FROM bestelling GROUP BY klant_idHoeveel bestellingen per klant?
SELECT categorie, COUNT(*) FROM product GROUP BY categorieAantal producten per categorie
SELECT stad, COUNT(*) FROM klant GROUP BY stadAantal klanten per stad
SELECT categorie, COUNT(*) FROM product GROUP BY categorie HAVING COUNT(*) > 1Alleen categorieën met meer dan 1 product
💡 WHERE filtert vóór GROUP BY · HAVING filtert ná GROUP BY (op het groepsresultaat)
JOIN Tabellen koppelen
-- Impliciete JOIN (komma-syntax): SELECT t1.kolom, t2.kolom FROM tabel1 t1, tabel2 t2 WHERE t1.fk = t2.pk;
QueryResultaat
SELECT k.naam, b.datum, b.status
FROM bestelling b, klant k
WHERE b.klant_id = k.klant_id		Elke bestelling met de naam van de klant erbij
SELECT k.naam, p.naam, b.datum
FROM bestelling b, klant k, product p
WHERE b.klant_id = k.klant_id
AND b.product_id = p.product_id		Klantnaam + productnaam + datum voor elke bestelling
SELECT k.naam, r.score, r.commentaar
FROM review r, klant k
WHERE r.klant_id = k.klant_id		Reviews met naam van de reviewer
💡 b, k, p zijn aliassen — kortere namen voor de tabel in de query. 💡 Koppel altijd via FK = PK: b.klant_id = k.klant_id
ORDER BY · LIMIT Sorteren en beperken
ORDER BY kolom ASC -- laag → hoog ORDER BY kolom DESC -- hoog → laag LIMIT n -- max n rijen
QueryResultaat
SELECT naam, prijs FROM product ORDER BY prijs ASCProducten van goedkoopst naar duurste
SELECT naam, prijs FROM product ORDER BY prijs DESCProducten van duurste naar goedkoopste
SELECT naam FROM klant ORDER BY naam ASC LIMIT 3De eerste 3 klanten alfabetisch
SELECT naam, stock FROM product ORDER BY stock ASC LIMIT 5Top 5 producten met laagste stock
💡 Combineer ORDER BY + LIMIT voor "Top N" queries (bijv. top 3 beste reviews).

JOIN ... ON Tabellen koppelen (ANSI-standaard)
SELECT t1.kolom, t2.kolom FROM tabel1 t1 INNER JOIN tabel2 t2 ON t1.id = t2.fk_id; -- Meerdere JOINs ketenen SELECT k.naam, p.naam, b.datum FROM bestelling b INNER JOIN klant k ON b.klant_id = k.klant_id INNER JOIN product p ON b.product_id = p.product_id;
JOIN-typeWat doet het?
INNER JOINEnkel rijen met een overeenkomst in beide tabellen
LEFT JOINAlle rijen uit de linker tabel, rechts NULL als geen match
RIGHT JOINAlle rijen uit de rechter tabel, links NULL als geen match
💡 ON specificeert de koppelconditie (FK = PK). Gebruik tabel-aliassen (k, b, p) om korter te schrijven. 💡 Meerdere JOINs ketenen: voeg per JOIN één extra tabel toe met een eigen ON-conditie.
GROUP BY · HAVING Groeperen en filteren op groepen
SELECT kolom, COUNT(*) -- of SUM, AVG, MAX, MIN FROM tabel GROUP BY kolom HAVING COUNT(*) > 1; -- filter ná groepering
QueryResultaat
SELECT categorie, SUM(prijs) FROM product GROUP BY categorieTotale prijs per categorie
SELECT status, COUNT(*) FROM bestelling GROUP BY status HAVING COUNT(*) > 1Enkel statussen met meer dan 1 bestelling
SELECT stad, COUNT(*) FROM klant GROUP BY stadAantal klanten per stad
⚠️ WHERE vs HAVING: WHERE filtert individuele rijen vóór groepering. HAVING filtert groepen na groepering.
NULL NULL-waarden afhandelen
-- NULL = afwezigheid van waarde WHERE kolom IS NULL -- heeft geen waarde WHERE kolom IS NOT NULL -- heeft wél een waarde -- COALESCE: geef eerste niet-NULL waarde COALESCE(kolom, 'Onbekend')
QueryResultaat
SELECT naam FROM klant WHERE stad IS NULLKlanten zonder stad ingevuld
SELECT naam FROM klant WHERE stad IS NOT NULLKlanten mét stad
SELECT naam, COALESCE(stad, 'Onbekend') AS stad FROM klantNULL vervangt door tekst 'Onbekend'
⚠️ WHERE stad = NULL werkt NOOIT. Gebruik altijd IS NULL. 💡 COALESCE(a, b, c) geeft de eerste niet-NULL waarde terug uit de lijst.
CASE WHEN Conditionele kolom berekenen
SELECT naam, CASE WHEN prijs < 20 THEN 'Goedkoop' WHEN prijs < 100 THEN 'Gemiddeld' ELSE 'Duur' END AS prijsklasse FROM product;
QueryResultaat
SELECT naam, CASE WHEN actief = 1 THEN 'Actief' ELSE 'Inactief' END AS status FROM klantLeesbaar label i.p.v. 0/1 voor elke klant
SELECT naam, CASE WHEN stock = 0 THEN 'Uitverkocht' WHEN stock < 5 THEN 'Bijna op' ELSE 'Op voorraad' END AS voorraad FROM productDrie mogelijke labels per product
💡 CASE WHEN werkt als if/else binnen SQL — ideaal voor labels, categorieën en tekstuele weergaven.
LIKE · IN · BETWEEN Flexibele WHERE-condities
-- LIKE: patroonherkenning WHERE naam LIKE '%Jan%' -- bevat 'Jan' WHERE email LIKE '%@gmail%' -- Gmail-adres -- IN: lijst van waarden WHERE stad IN ('Gent', 'Brussel') -- BETWEEN: bereik (inclusief) WHERE prijs BETWEEN 10 AND 50
PatroonBetekenis
%tekst%Bevat 'tekst' ergens
tekst%Begint met 'tekst'
%tekstEindigt op 'tekst'
_Precies één willekeurig teken
💡 IN ('a','b','c') is korter dan drie OR-condities schrijven. 💡 BETWEEN a AND b is inclusief: a en b zelf worden ook meegenomen.
SUBQUERY Query binnen een query
-- Subquery in WHERE SELECT naam, prijs FROM product WHERE prijs > (SELECT AVG(prijs) FROM product); -- Subquery met IN SELECT naam FROM klant WHERE klant_id IN (SELECT klant_id FROM bestelling); -- Subquery met NOT IN SELECT naam FROM klant WHERE klant_id NOT IN (SELECT klant_id FROM bestelling);
GebruikWat doet het?
WHERE prijs > (SELECT AVG(prijs) FROM product)Producten duurder dan het gemiddelde
WHERE klant_id IN (SELECT klant_id FROM bestelling)Klanten die minstens één bestelling hebben
WHERE klant_id NOT IN (SELECT klant_id FROM bestelling)Klanten die nog NOOIT besteld hebben
💡 De binnenste query wordt eerst uitgevoerd, dan pas de buitenste. 💡 Gebruik IN voor "zit in de groep" en NOT IN voor "zit er NIET in".
DISTINCT Unieke waarden opvragen
-- Verwijder dubbele rijen uit het resultaat SELECT DISTINCT kolom FROM tabel; -- Tellen hoeveel unieke waarden SELECT COUNT(DISTINCT kolom) FROM tabel;
QueryResultaat
SELECT DISTINCT stad FROM klantAlle unieke steden (elke stad maar één keer)
SELECT DISTINCT categorie FROM productAlle unieke productcategorieën
SELECT DISTINCT status FROM bestellingWelke statussen bestaan er?
SELECT COUNT(DISTINCT stad) FROM klantHoeveel verschillende steden zijn er?
💡 DISTINCT staat vlak na SELECT — het filtert dubbele rijen uit het eindresultaat. 💡 Combineer met COUNT(DISTINCT ...) om unieke aantallen te tellen.
AS Aliassen voor kolommen en tabellen
-- Kolom een andere naam geven SELECT prijs AS verkoopprijs FROM product; -- Tabel een korte naam geven SELECT k.naam, b.datum FROM klant AS k JOIN bestelling AS b ON k.klant_id = b.klant_id; -- Berekende kolom een naam geven SELECT naam, prijs * 1.21 AS prijs_incl_btw FROM product;
QueryResultaat
SELECT naam AS klantnaam FROM klantKolom heet "klantnaam" in het resultaat
SELECT COUNT(*) AS aantal_klanten FROM klantTeller krijgt een leesbare naam
SELECT AVG(prijs) AS gemiddelde_prijs FROM productGemiddelde prijs met duidelijke kolomnaam
SELECT prijs * stock AS voorraadwaarde FROM productBerekening met een eigen naam
💡 AS is optioneel — SELECT naam klantnaam werkt ook, maar AS is leesbaarder. 💡 Tabelaliassen (k, b, p) zijn essentieel bij JOINs om onduidelijkheid over kolomnamen te vermijden.
LEFT JOIN Alle rijen uit de linker tabel, ook zonder match
-- INNER JOIN: enkel matches SELECT k.naam, b.datum FROM klant k INNER JOIN bestelling b ON k.klant_id = b.klant_id; -- LEFT JOIN: alle klanten, ook zonder bestelling SELECT k.naam, b.datum FROM klant k LEFT JOIN bestelling b ON k.klant_id = b.klant_id;
JOIN-typeResultaat
INNER JOINEnkel klanten MET minstens één bestelling
LEFT JOINAlle klanten — datum is NULL als er geen bestelling is
💡 Gebruik LEFT JOIN + WHERE b.klant_id IS NULL om klanten te vinden die NOOIT besteld hebben. ⚠️ De tabel na FROM is "links", de tabel na LEFT JOIN is "rechts". Alle linkerrijen blijven altijd zichtbaar.
